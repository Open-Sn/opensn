   #pragma once
   
   #include "framework/math/quadratures/angular/angular_quadrature.h"
   #include <vector>
   #include <functional>
   
   namespace opensn
   {
   
   /// Rule-based harmonic selection for different quadrature types.
   ///
   /// Implements the specific mathematical rules for selecting spherical harmonics
   /// based on quadrature type, dimension, and operator construction method.
   class HarmonicSelectionRules
   {
   public:
     /// Input parameters controlling harmonic selection.
     struct SelectionParameters
     {
       AngularQuadratureType quadrature_type{};
       OperatorConstructionMethod construction_method{};
       unsigned int dimension = 0;
       unsigned int scattering_order = 0;
       size_t num_angles = 0;
   
       /// Quadrature-specific order parameter.
       /// For Lebedev: Lebedev order (3, 5, 7, 9, etc.)
       /// For SLDFESQ: Uniform Refinement level
       unsigned int quadrature_order = 0;
   
       /// Number of positive-mu polar points (product and triangular quadratures).
       unsigned int n_polar = 0;
       /// Total number of azimuthal points (product and triangular quadratures).
       unsigned int n_azimuthal = 0;
     };
   
     /// Select harmonics using predetermined mathematical rules.
     ///
     /// \param params Selection parameters including quadrature type and dimensions.
     /// \return Vector of selected harmonic indices.
     static std::vector<AngularQuadrature::HarmonicIndices>
     SelectHarmonics(const SelectionParameters& params);
   
   private:
     /// Select harmonics for a 2D Cartesian-product quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     Select2DCartesianProduct(const SelectionParameters& params);
   
     /// Select harmonics for a 3D Cartesian-product quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     Select3DCartesianProduct(const SelectionParameters& params);
   
     /// Select harmonics for a 2D triangular quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     Select2DTriangular(const SelectionParameters& params);
   
     /// Select harmonics for a 3D triangular quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     Select3DTriangular(const SelectionParameters& params);
   
     /// Select harmonics for a Lebedev quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     SelectLebedev(const SelectionParameters& params);
   
     /// Select harmonics for an SLDFESQ quadrature (Galerkin rule).
     static std::vector<AngularQuadrature::HarmonicIndices>
     SelectSLDFESQ(const SelectionParameters& params);
   
     /// Map number of Lebedev quadrature points to Lebedev order.
     static int DetermineLebedevOrder(size_t num_angles, unsigned int dimension = 3);
   
     /// Map number of SLDFESQ quadrature points to refinement level.
     static int DetermineSLDFELevel(size_t num_angles, unsigned int dimension = 3);
   
     /// Select harmonics for a 1D quadrature (all construction methods).
     static std::vector<AngularQuadrature::HarmonicIndices>
     Select1D(const SelectionParameters& params);
   
     /// Standard (non-Galerkin) harmonic selection fallback.
     static std::vector<AngularQuadrature::HarmonicIndices>
     SelectStandard(const SelectionParameters& params);
   };
   
   } // namespace opensn
